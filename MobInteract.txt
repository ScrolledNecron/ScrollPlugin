@EventHandler
    public void onInteract(PlayerInteractEntityEvent e) {
        Player p = e.getPlayer();
        Location pLoc = p.getLocation();
        Entity entity = e.getRightClicked();
        if (entity.getCustomName().contains("Helper")){

            //Message 1
            new BukkitRunnable(){

                @Override
                public void run() {
                    p.sendMessage(ChatColor.GREEN + "Helper " + ChatColor.DARK_GRAY + ": " + ChatColor.YELLOW + "Greetings " + p.getDisplayName() + " I have been here for a long time...");
                    p.playSound(pLoc, Sound.ENTITY_WITHER_SHOOT, 1,20);
                }
            }.runTaskLater(ScrollPlugin.getInstance(),40);

            //Message 2
            new BukkitRunnable(){

                @Override
                public void run() {
                    p.sendMessage(ChatColor.GREEN + "Helper " + ChatColor.DARK_GRAY + ": " + ChatColor.YELLOW + "You might have come here to get challenges, but beware I warn you this place is no joke for irregular peoples.");
                    p.playSound(pLoc, Sound.ENTITY_WITHER_SHOOT,1,20);
                }
            }.runTaskLater(ScrollPlugin.getInstance(),40);

            //Message 3
            new BukkitRunnable(){

                @Override
                public void run() {
                    p.sendMessage(ChatColor.GREEN + "Helper " + ChatColor.DARK_GRAY + ": " + ChatColor.YELLOW + "Well, if you really want to be " + ChatColor.RED + "challenged " + ChatColor.YELLOW + "then speak to the guard he will guide you.");
                    p.playSound(pLoc,Sound.ENTITY_WITHER_SHOOT,1,20);
                }
            }.runTaskLater(ScrollPlugin.getInstance(),30);

            System.out.println("Tasks success!");
        }else {
            return;
        }
    }
